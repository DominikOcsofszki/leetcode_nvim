class Solution:
    def isValid(self, s: str) -> bool:
        arr = []
        dic = {}
        dic['('] = ')'
        dic['{'] = '}'
        dic['['] = ']'
        for bra in s:
            if arr and bra == dic.get(arr[-1]):
                arr.pop()
            else:
                arr.append(bra)
        return len(arr) == 0
