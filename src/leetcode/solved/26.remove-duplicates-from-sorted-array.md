class Solution:
    def removeDuplicates(self, nums: List[int]) -> int:
        check = {}
        lst = []
        for i, item in enumerate(nums):
            if item not in check:
                check[item] = item
                lst.append(i)

        unique = len(lst)
        for i, i_item in enumerate(lst):
            nums[i] = nums[i_item]
        return unique
